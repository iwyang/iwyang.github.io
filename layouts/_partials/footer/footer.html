=========================================================== */}} {{- $gitHashFull := "" -}} {{- $gitHashShort := "" -}} {{- $lastBuildTime := now -}} {{/* --- 1. 获取 Git Hash --- */}} {{- if fileExists ".git/HEAD" -}} {{- $head := readFile ".git/HEAD" | strings.TrimSpace -}} {{- if strings.HasPrefix $head "ref: " -}} {{- $ref := strings.TrimPrefix "ref: " $head -}} {{- if fileExists (printf ".git/%s" $ref) -}} {{- $gitHashFull = readFile (printf ".git/%s" $ref) | strings.TrimSpace -}} {{- end -}} {{- else -}} {{- $gitHashFull = $head -}} {{- end -}} {{- end -}} {{- if eq $gitHashFull "" -}} {{- if getenv "GITHUB_SHA" -}} {{- $gitHashFull = getenv "GITHUB_SHA" -}} {{- else if getenv "COMMIT_REF" -}} {{- $gitHashFull = getenv "COMMIT_REF" -}} {{- end -}} {{- end -}} {{/* --- 2. 使用 Hash 请求 GitHub API (最新 try 语法) --- */}} {{- if ne $gitHashFull "" -}} {{- $repo := "iwyang/iwyang.github.io" -}} {{- $apiURL := printf "https://api.github.com/repos/%s/commits/%s" $repo $gitHashFull -}} {{- $ghToken := getenv "GITHUB_TOKEN" -}} {{- $resp := "" -}} {{/* 根据是否有 Token 发起带 try 的请求 */}} {{- if ne $ghToken "" -}} {{- $opts := dict "headers" (dict "Authorization" (printf "Bearer %s" $ghToken)) -}} {{- $resp = try (resources.GetRemote $apiURL $opts) -}} {{- else -}} {{- $resp = try (resources.GetRemote $apiURL) -}} {{- end -}} {{/* 处理 try 的返回结果 */}} {{- with $resp -}} {{- if .Err -}} {{/* 如果请求失败，打印警告并走兜底时间 */}} {{- warnf "获取 Commit 时间失败: %s" .Err -}} {{- else -}} {{/* 请求成功，.Value 即为获取到的资源 */}} {{- $apiData := .Value -}} {{- if $apiData -}} {{- $json := $apiData.Content | transform.Unmarshal -}} {{- if and $json $json.commit -}} {{- $lastBuildTime = time.AsTime $json.commit.committer.date -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{/* 截取短 Hash */}} {{- if ge (len $gitHashFull) 7 -}} {{- $gitHashShort = substr $gitHashFull 0 7 -}} {{- end -}} {{- end -}} {{/* =========================================================== 【页面展示】
